CREATE OR REPLACE TRIGGER SELL_LOGS
AFTER INSERT OR UPDATE ON SELL

DECLARE
OPNAME VARCHAR2(50);

BEGIN
IF INSERTING THEN
OPNAME := 'INSERT';

ELSIF UPDATING THEN
OPNAME := 'UPDATE';

END IF;
INSERT INTO SELL_LOG VALUES(USER, OPNAME, SYSDATE);
END;
